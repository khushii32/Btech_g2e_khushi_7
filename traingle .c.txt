#include <stdio.h>

int main() {
float a, b, c;

printf("Enter three sides of triangle: ");
scanf("%f %f %f", &a, &b, &c);

if (a <= 0 || b <= 0 || c <= 0) {
printf("Sides must be positive numbers.\n");
}
else if ((a + b > c) && (a + c > b) && (b + c > a)) {
printf("The given sides form a valid triangle.\n");
}
else {
printf("The given sides do NOT form a valid triangle.\n");
}

return 0;
}
